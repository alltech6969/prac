public class WebService1 : System.Web.Services.WebService
 {

     [WebMethod]
     public double D2R(double dollar)
     {
         double ans = dollar * 82;
         return ans;
     }

     [WebMethod]
     public double R2D(double rupees)
     {
         double ans = rupees / 82;

Pr 5: KVM

Steps:
	1. 	Go to CC_ISO's Extract the Ubuntu 22.04 Zip and double click the blue(.vbox) file 
	2.	Launch the image in Virtual box.
	3.	Launch Terminal and execute the following code.
	4.	Update and Upgrade Ubuntu 22.04 
		sudo apt update 
		sudo apt upgrade 
	5.	Check if Virtualization is enabled 
		egrep -c '(vmx|svm)' /proc/cpuinfo 
	6.	Install the cpu-checker package 
		sudo apt install -y cpu-checker 
	7.	verify if KVM virtualization is enabled 
		kvm-ok 
	8.	Install KVM on Ubuntu 22.04 
		sudo apt install -y qemu-kvm virt-manager libvirt-daemon-system virtinst libvirt-clients bridge-utils 
	9.	Enable the virtualization daemon 
		sudo systemctl enable --now libvirtd sudo systemctl start libvirtd 
	10.	Check virtualization daemon is running 
		sudo systemctl status libvirtd 
	11.	Add Your User to the KVM and Libvirt Group 
		sudo usermod -aG kvm $USER 
		sudo usermod -aG libvirt $USER
	12.	Restart the machine.
	13.	Search for Virtual Machine manager and launch it.

	6. Click Add
	7. in the opened file add the code below HelloWorld() method
		[WebMethod, Description("Get Image Content")]
		public byte[] GetImageFile(String fileName)
		{
			string filePath = Server.MapPath("~/Images/") + fileName;

			if(System.IO.File.Exists(filePath))
			{
				return System.IO.File.ReadAllBytes(filePath);
			}
			else
			{
				return new byte[] { 0 };
			}
		}
	
	8. Add a new Generic Handler (Project > Add New Item > Generic Handler)
	9. Give Name : ImageHandler.ashx
	10. Modify the following code on the opened file.
		public void ProcessRequest(HttpContext context)
		{
			ImageDownloadWebService imageDownloadWebService = new ImageDownloadWebService();

			byte[] binaryImage = imageDownloadWebService.GetImageFile(context.Request["fileName"]);

			if(binaryImage.Length == 1)
			{
				//no content, do nothing
			}
			else 
			{
				context.Response.ContentType = "image/jpeg";

				context.Response.BinaryWrite(binaryImage);
			}
		}

Steps:
	1.	Go to CC_ISO's Extract the Ubuntu 22.04 Zip and double click the blue(.vbox) file 
	2.	Launch the image in Virtual box.
	3.	Launch Terminal and execute the following code.
	4.	sudo useradd -s /bin/bash -d /opt/stack -m stack
		This code will create a new user stack with its home directory being /opt/stack
	5.	sudo chmod +x /opt/stack
		Adding execute permission to /opt/stack
	6.	echo "stack ALL=(ALL) NOPASSWD: ALL" | sudo tee /etc/sudoers.d/stack
	7.	sudo -u stack -i
		the commands 6 & 7 makes sure that stack user has sudo previlages and doesnot have to enter password again and again.
	8.	git clone https://opendev.org/openstack/devstack
	9.	cd devstack
		Downloading devstack and changing directory to devstack
	10.	 Now we have create a file and add the following lines to it.
	11.	nano local.conf 
		(this will create and open the local.conf file)

		[[local|localrc]]
		ADMIN_PASSWORD=secret
		DATABASE_PASSWORD=$ADMIN_PASSWORD
		RABBIT_PASSWORD=$ADMIN_PASSWORD
		SERVICE_PASSWORD=$ADMIN_PASSWORD
